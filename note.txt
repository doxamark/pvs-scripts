SELECT BillID
INTO TABLE TempBillIDs
FROM Bill
JOIN BillStatus on Bill.Status = BillStatus.Code and BillStatus.Updateable = 0
JOIN Parcel on Bill.ParcelID = Parcel.ParcelID
WHERE Bill.SysYear = 2024
AND Parcel.REPP = 'P'
AND Bill.CollectorID = 568;
 
UPDATE Bill
JOIN BillStatus on Bill.Status = BillStatus.Code and BillStatus.Updateable = 0
JOIN Parcel on Bill.ParcelID = Parcel.ParcelID
SET Bill.UpdateOverride = 1
WHERE Bill.SysYear = 2024
AND Parcel.REPP = 'P'
AND Bill.CollectorID = 568;
 
UPDATE Bill
JOIN Parcel on Bill.ParcelID = Parcel.ParcelID
SET Bill.AccountLookup = left(REPLACE(Bill.Account, ' ', ''),5)+right(REPLACE(Bill.Account, ' ', ''),8)
WHERE Bill.SysYear = 2024
AND Parcel.REPP = 'P'
AND Bill.CollectorID = 568;
 
UPDATE Bill
JOIN TempBillIDS on Bill.BillID = TempBillIDs.BillID
SET Bill.UpdateOverride = 0;
 
DROP TABLE TempBillIDs;
